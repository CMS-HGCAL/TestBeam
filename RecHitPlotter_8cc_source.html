177 RecHitPlotter::analyze(const edm::Event& event, const edm::EventSetup& setup)
178 {
179 
180  using namespace edm;
181  using namespace std;
182 
183  edm::Handle<HGCalTBRecHitCollection> Rechits;
184  event.getByToken(HGCalTBRecHitCollection_, Rechits);
185 
186  for(auto RecHit : *Rechits) {
187  if(!IsCellValid.iu_iv_valid((RecHit.id()).layer(), (RecHit.id()).sensorIU(), (RecHit.id()).sensorIV(), (RecHit.id()).iu(), (RecHit.id()).iv(), sensorsize)) continue;
188  int n_layer = (RecHit.id()).layer();
189  int n_cell_type = (RecHit.id()).cellType();
190  double n_cell_area = IsCellValid.Cell_Area(n_cell_type);
191 //We now obtain the cartesian coordinates of the cell corresponding to an iu,iv. This may either be a full hex, a half hex or an invalid cell. If a cell is invalid based on the iu,iv index -123456 is returned for its x,y vertices
192  CellCentreXY = TheCell.GetCellCentreCoordinatesForPlots((RecHit.id()).layer(), (RecHit.id()).sensorIU(), (RecHit.id()).sensorIV(), (RecHit.id()).iu(), (RecHit.id()).iv(), sensorsize);
193  double iux = (CellCentreXY.first < 0 ) ? (CellCentreXY.first + delta) : (CellCentreXY.first - delta) ;
194  double iyy = (CellCentreXY.second < 0 ) ? (CellCentreXY.second + delta) : (CellCentreXY.second - delta);
195  h_RecHit_layer[n_layer - 1]->Fill(iux , iyy, RecHit.energy());
196  h_RecHit_layer_summed[n_layer - 1]->Fill(RecHit.energy());
197 //The energy threshold for the occupancy has been hardcoded here. Need to decide what a good choice is. Maybe dynamic per cell depending on the pedestal
198  if(RecHit.energy() > 5) h_RecHit_layer_Occupancy[n_layer - 1]->Fill(iux , iyy, 1./n_cell_area);
199 // There will be several array indices iu, iv that wont be filled due to it being invalid. Can think of alternate array filling.
200  h_RecHit_layer_cell[n_layer - 1][7 + (RecHit.id()).iu()][7 + (RecHit.id()).iv()]->Fill(RecHit.energy());
201  }
