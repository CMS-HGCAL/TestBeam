115 {
116  using namespace edm;

118  edm::Handle<HGCalTBDigiCollection> digisHandle;

121  HGCalCondPedestals pedestals_cond(HGCalTBNumberingScheme::scheme(), 0);
122  HGCalCondNoise noise_cond(HGCalTBNumberingScheme::scheme(), 0);
123 
124  HGCalCondObjectTextIO condIO(HGCalTBNumberingScheme::scheme());
125  HGCalElectronicsMap emap;
126 
127  condIO.load("CondObjects/data/map_FNAL_2.txt", emap);
128 
129  for(auto digi_itr = digisHandle->begin(); digi_itr != digisHandle->end(); ++digi_itr) {
130 
131  SKIROC2DataFrame digi(*digi_itr);
132  HGCalTBDetId detId = digi.detid();
133 
134  unsigned int nSamples = digi.samples();
135 
136  /**** WARNING \todo multiple samples are not yet allowed */
137  nSamples=1;
138  /**** */
139 
140  pedestalMap_t::iterator thisPedestal_itr = pedestals.find(detId);
141  if(thisPedestal_itr==pedestals.end()){
142  pedestalSum_t empty_ped;
143  empty_ped.sum=0.;
144  empty_ped.n=0;
145 
146  pedestals[detId] = empty_ped;
147  thisPedestal_itr = pedestals.find(detId);
148  assert(thisPedestal_itr!=pedestals.end());
149  }
150 
151  // now sum the values from different events
152  for(unsigned int iSample = 0; iSample < nSamples; ++iSample) {
153  thisPedestal_itr->second.sum += digi[iSample].adcLow();
154  thisPedestal_itr->second.sum2 += digi[iSample].adcLow() * digi[iSample].adcLow();
155  ++(thisPedestal_itr->second.n);
156 #ifdef DEBUG
157  std::cout << "[PEDESTAL PRODUCER: digi]" << *digi_itr << std::endl;
158  std::cout << " ** " << thisPedestal_itr->second.sum << std::endl;
159 #endif
160 
161  }
162  }
163 
164  // now I have the pedestal average, I have to print them
165  for(const auto pedestal : pedestals){
166  float pedestal_mean = pedestal.second.sum/pedestal.second.n;
167  pedestals_cond.set(pedestal.first, pedestal_mean);
168  noise_cond.set(pedestal.first, sqrt(pedestal.second.sum2/pedestal.second.n - (pedestal_mean * pedestal_mean)));
169  }
170  condIO.store("newPedestals.txt", pedestals_cond);
171  condIO.store("newNoise.txt", noise_cond);
